/** * Cloudflare Worker - 台股收盤價查詢 * * 架構說明： * 改用 TWSE MIS 即時報價 API（mis.twse.com.tw） * 此 API 上市用 tse_代號.tw，上櫃用 otc_代號.tw * 一次可查多支，用 | 分隔 * * GET /api/quotes?codes=2330,00679B,0050,3390 * GET /ping */ const MIS_URL = 'https://mis.twse.com.tw/stock/api/getStockInfo.jsp'; const CORS = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, OPTIONS', 'Content-Type': 'application/json; charset=utf-8', }; // 上櫃代號特徵：純數字4碼但屬於OTC、或含英文字母(如00679B) // 策略：先試 tse_，失敗或無資料再試 otc_ // 實務上：用 | 把 tse 和 otc 兩個都送進去，API 會回傳有資料的那個 export default { async fetch(request) { const url = new URL(request.url); if (request.method === 'OPTIONS') { return new Response(null, { status: 204, headers: CORS }); } if (url.pathname === '/ping') { return new Response(JSON.stringify({ ok: true, time: new Date().toISOString() }), { headers: CORS }); } if (url.pathname === '/api/quotes') { const codesParam = url.searchParams.get('codes') || ''; const codes = codesParam.split(',').map(c => c.trim().toUpperCase()).filter(Boolean); if (!codes.length) { return new Response(JSON.stringify({ error: 'codes param required' }), { status: 400, headers: CORS }); } return queryStocks(codes); } return new Response( JSON.stringify({ error: 'Not found', usage: '/api/quotes?codes=2330,00679B,0050' }), { status: 404, headers: CORS } ); } }; async function queryStocks(codes) { // 對每個代號，同時送 tse_ 和 otc_ 版本，讓 API 自己回有資料的 const exCodes = codes.flatMap(c => [`tse_${c}.tw`, `otc_${c}.tw`]); const param = exCodes.join('|'); const apiUrl = `${MIS_URL}?ex_ch=${encodeURIComponent(param)}&json=1&delay=0`; try { const res = await fetch(apiUrl, { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36', 'Referer': 'https://mis.twse.com.tw/', 'Accept': 'application/json', }, signal: AbortSignal.timeout(10000), }); if (!res.ok) { return new Response( JSON.stringify({ error: `MIS API returned ${res.status}` }), { status: 502, headers: CORS } ); } const data = await res.json(); const msgArray = data?.msgArray || []; // 整理成 { 代號: { name, close, change, source } } const result = {}; msgArray.forEach(item => { const code = (item.c || '').trim(); const name = (item.n || '').trim(); const ex = (item.ex || '').trim(); // tse 或 otc // z = 最新成交價（盤中）或收盤價 // y = 昨收 const close = parseFloat(item.z) || parseFloat(item.y) || 0; const prevClose = parseFloat(item.y) || 0; const change = prevClose ? +(close - prevClose).toFixed(2) : 0; if (code && close) { result[code] = { name, close: close.toFixed(2), change: change.toFixed(2), source: ex === 'tse' ? 'TWSE' : 'OTC', }; } }); return new Response(JSON.stringify(result), { headers: CORS }); } catch (e) { return new Response( JSON.stringify({ error: e.message }), { status: 500, headers: CORS } ); } }
